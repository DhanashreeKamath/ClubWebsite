#include <iostream>
using namespace std;

int main() {

    int number = 10;
    int sum = 0;
    int average = 0;
    cout << "Print First 10 numbers and its average :\n";
    for (int i=0;i<number;i++)
    {
       sum = sum + i;
    }
    average = sum/2;
    cout << "Sum =%d"<< sum;
    cout << "Average = %d"<< average;
    return 0;
}
